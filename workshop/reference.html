2Risk score

"Identify potentially fraudulent transactions based on amount, login attempts, and duration."

SQL
SELECT "AccountID", "TransactionID", "TransactionAmount",
       CAST("LoginAttempts" AS DECIMAL) AS "LOGIN_ATTEMPTS",
       "TransactionDuration", "Channel", "TransactionDate",
       (CASE WHEN "TransactionAmount" >= 881 THEN 1 ELSE 0 END
      + CASE WHEN CAST("LoginAttempts" AS DECIMAL) >= 4 THEN 1 ELSE 0 END
      + CASE WHEN "TransactionDuration" >= 260 THEN 1 ELSE 0 END) AS "RISK_FLAGS"
FROM "STARTER_KIT"."BANK_TRANSACTIONS_DATA"
WHERE (CASE WHEN "TransactionAmount" >= 881 THEN 1 ELSE 0 END
     + CASE WHEN CAST("LoginAttempts" AS DECIMAL) >= 4 THEN 1 ELSE 0 END
     + CASE WHEN "TransactionDuration" >= 260 THEN 1 ELSE 0 END) >= 2
ORDER BY "RISK_FLAGS" DESC, "TransactionAmount" DESC;
3Top locations

"Show the top 5 locations by total transaction volume."

SQL
SELECT "Location", COUNT(*) AS "TXN_COUNT",
       ROUND(SUM("TransactionAmount"), 2) AS "TOTAL_VALUE"
FROM "STARTER_KIT"."BANK_TRANSACTIONS_DATA"
GROUP BY "Location"
ORDER BY "TOTAL_VALUE" DESC
LIMIT 5;
4Channel value

"Which transaction channels generate the highest transaction value?"

SQL
SELECT "Channel", COUNT(*) AS "TXN_COUNT",
       ROUND(SUM("TransactionAmount"),2) AS "TOTAL_VALUE",
       ROUND(AVG("TransactionAmount"),2) AS "AVG_VALUE"
FROM "STARTER_KIT"."BANK_TRANSACTIONS_DATA"
GROUP BY "Channel"
ORDER BY "TOTAL_VALUE" DESC;
5Spend by occupation

"Which customer occupations have the highest average transaction amount?"

SQL
SELECT "CustomerOccupation", COUNT(*) AS "TXN_COUNT",
       ROUND(AVG("TransactionAmount"),2) AS "AVG_AMOUNT"
FROM "STARTER_KIT"."BANK_TRANSACTIONS_DATA"
GROUP BY "CustomerOccupation"
ORDER BY "AVG_AMOUNT" DESC;
